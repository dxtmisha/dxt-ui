 */

    item: ShallowRef<T | undefined, T | undefined>;

/**
 * Manages selected frames in Figma UI.
 * @returns Selected IDs, loading state, and utilities
 */
export declare function useFigmaUiSelected(): {
    /** Selected frame IDs */
    selected: ComputedRef<string[]>;
    /** Loading state */
    loading: ComputedRef<boolean>;
    /**
     * Check if frame is selected
     * @param id Frame ID
     * @returns Boolean ref
     */
    isSelected(id: string): ComputedRef<boolean | undefined>;
    /**
     * Toggle frame selection
     * @param id Frame ID
     * @param selected Selection state
     */
    toggleSelected(id: string, selected: boolean): void;
};
